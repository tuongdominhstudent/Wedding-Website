**SectionBand.jsx**

```jsx
import styles from './SectionBand.module.css';

const SEPARATOR = '\u00A0\u00A0\u25C6\u00A0\u00A0'; // ◆ with spacing
const REPEAT_COUNT = 6;

function buildTrackText(text) {
  return Array.from({ length: REPEAT_COUNT }, () => text).join(SEPARATOR) + SEPARATOR;
}

function SectionBand({ text }) {
  const trackText = buildTrackText(text);
  return (
    <div className={styles.band} aria-hidden="true">
      <div className={styles.track}>
        <span className={styles.item}>{trackText}</span>
        <span className={styles.item} aria-hidden="true">{trackText}</span>
      </div>
    </div>
  );
}

export default SectionBand;
```

Two identical `<span>` elements side by side. The CSS animation shifts the outer track by -50%, which moves the first span fully off screen while the second (identical) span takes its place — seamless loop.

**SectionBand.module.css**

```css
.band {
  width: 100%;
  height: 72px;
  background-color: #ffffff;
  border-top: 1px solid rgba(0, 0, 0, 0.12);
  border-bottom: 1px solid rgba(0, 0, 0, 0.12);
  overflow: hidden;
  display: flex;
  align-items: center;
  position: relative;
  z-index: var(--z-content);
}

.track {
  display: flex;
  flex-shrink: 0;
  white-space: nowrap;
  animation: marquee 28s linear infinite;
  will-change: transform;
}

.item {
  display: inline-block;
  font-family: var(--font-family-display);
  font-size: 13px;
  font-weight: 400;
  letter-spacing: 0.25em;
  text-transform: uppercase;
  color: #000000;
  padding-right: 0; /* spacing handled by SEPARATOR in JS */
}

@keyframes marquee {
  from {
    transform: translateX(0);
  }
  to {
    transform: translateX(-50%);
  }
}

@media (prefers-reduced-motion: reduce) {
  .track {
    animation-play-state: paused;
  }
}
```
